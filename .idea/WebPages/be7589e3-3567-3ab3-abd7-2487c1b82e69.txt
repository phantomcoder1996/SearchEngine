}

}

viewport = _getWindowViewport();
}

viewport.right >= Math.max(pos.left + offset.left, 0) &&
viewport.top <= Math.max(pos.top + elmt.height() + offset.bottom, 0) &&
viewport.left <= Math.max(pos.left + elmt.width() + offset.right, 0);

return false;
}

});
if (!wasInView && t.inView) {

  function getResultUrl(selectedResult) {

      var length = selectedResult.urls.length;
        for (var i = 0; i < length; i++) {
        var url = selectedResult.urls[i];
        if (url.url_type === linkType) {
          return url.url || url.fallback_url;

                      longitude: selectedResult.longitude,

      } else {
        return null;
      }
    } else if (selectedResult.geonavi) {
      return '/' + dataHolder.data('widgetname') + '-g' + (selectedResult.value || selectedResult.id);
    }

    // pass search session id into srp for tracking purpose
    if (selectedResult.type == 'SEARCH') {
        params = {
            uiOrigin: typeaheadTracking.getWidgetName(),
            searchSessionId: typeaheadTracking.getSearchSessionId()
        };

        var paramString = $.param(params);
        selectedResult.url = selectedResult.url + '&' + paramString;
    }

    return selectedResult.url;
  }

  function setDates(params) {
    var inDate = getDate('CHECKIN');
    if (inDate && inDate[0]) {
      params.inDay = inDate[0].getDate();
      params.inMonth = inDate[0].getMonth();

      params.outDay = outDate[0].getDate();
      params.outMonth = outDate[0].getMonth();

    }
  }

      ta.page &&

    }
    return null;
  }

  function getFirstValidResult(results, typeMustBeGeo) {
    // typeMustBeGeo is used for showing recent results on pageload. We show the parent geo of a hotel, etc. instead of the POI itself.

    if (results && results.length) {
      for (var i = 0; i < results.length; i++) {
        var result = results[i];
        var acceptableGeoType = !typeMustBeGeo || (typeMustBeGeo && result.type && result.type === 'GEO');
        if (result.url &&
          result.name &&
          acceptableGeoType) {
          return result;
        }
      }
    }
    return null;
  }

  function _detectContext() {
    var classArray = _widgetDiv.className.toLowerCase()
      .split(' ')
      .filter(function(s) { return s.indexOf('wctx-',0) === 0; });
    if (classArray.length > 0) {
      return classArray[0].substring(5);
    }
  }

  function _init() {
    _trackContext = _detectContext();
  }

  _init();

  return {};

define('utils/ResponsiveEvents', ['mixins/Events', 'utils/responsive', 'utils/throttle', 'vanillajs'],
function (Events, Responsive, Throttle) {
'use strict';
var lastWidth = document.documentElement.clientWidth;
var lastSizes = Responsive.currentBreakpoints();
var events = ['breakpoint'];
Responsive.breakpoints.forEach(function(size) {
events.push('over-' + size, 'under-' + size);
});
var delegate = Events.create(events);
delegate.onOver = function(size, fn) {
delegate.on('over-' + size, fn);
};
delegate.onUnder = function(size, fn) {
delegate.on('under-' + size, fn);
};
delegate.offOver = function(size, fn) {
delegate.off('over-' + size, fn);
};
delegate.offUnder = function(size, fn) {
delegate.off('under-' + size, fn);
};
function getChangedSizes(a, b, reverse) {
var aDiff = a.filter(function(x) { return b.indexOf(x) < 0; });
var bDiff = b.filter(function(x) { return a.indexOf(x) < 0; });
return reverse ? aDiff.reverse().concat(bDiff.reverse()) : aDiff.concat(bDiff);
}
function checkBreakpoints() {
var width = document.body.clientWidth;
if (width == lastWidth) {
return;
}
var expanded = width > lastWidth;
var sizes = Responsive.currentBreakpoints();
var diff = getChangedSizes(lastSizes, sizes, expanded);
diff.forEach(function(size) {
var prefix = expanded ? 'over-' : 'under-';
delegate.emit(prefix + size);
});
if (diff.length) {
delegate.emit('breakpoint', sizes[0]);
}
lastSizes = sizes;
lastWidth = width;
}
window.addEventListener('resize', Throttle(checkBreakpoints, 100));
return delegate;
});
