      <h2> Tag synonyms for <a href="/questions/tagged/cgi" class="post-tag" title="show questions tagged 'cgi'" rel="tag">cgi</a> </h2> 
      <div class="post-text"> 
       <p> Incorrectly tagged questions are hard to find and answer. If you know of <b>common, alternate spellings or phrasings</b> for this tag, add them here so we can automatically correct them in the future. For example, suggest “bike” as a synonym for bicycle, or “sock” for socks. </p> 
      </div> 
      <p> <span class="post-text"><a href="/questions/tagged/cgi" class="post-tag" title="show questions tagged 'cgi'" rel="tag">cgi</a> currently has no approved synonyms.</span> </p> 
      <p class="post-text"> <a href="/tags/synonyms" title="browse and sort through all tag synonyms">see all tag synonyms »</a> </p> 
      <br> 
      <div class="post-text"> 
       <p> Users with more than 2500 reputation and a total answer score of 5 or more on the tag, can <b>suggest</b> tag synonyms. Users with a total answer score (total upvotes minus total downvotes) of 5 or more on the tag, can <b>vote</b> for tag synonyms. Suggestions will be automatically approved when they reach a score of 4, and automatically deleted when they reach a score of -2. </p> 
      </div>

      <script>

      function updateTooltips() {
          var table = $('.synonyms table');
          $('.synonyms table td.voting a.up').attr('title', table.data('text-up'));
          $('.synonyms table td.voting a.down').attr('title', table.data('text-down'));
          $('.synonyms table td.voting a.voted').attr('title', table.data('text-back'));
      }

      function getId(node) {
          return parseInt($(node).parents('tr').attr('class').match(/\d+/)[0], 10);
      }

      function applyResult(result, tr) {
          tr.find('.score').text(result.score);
          if (result.upvoted) {
              tr.find('.up').addClass('voted');
          } else {
              tr.find('.up').removeClass('voted');
          }

          if (result.downvoted) {
              tr.find('.down').addClass('voted');
          } else {
              tr.find('.down').removeClass('voted');
          }

          updateTooltips();

      }
      StackExchange.ready(function () {

          if (typeof (bindTagFilterAutoComplete) == 'function') {
              bindTagFilterAutoComplete('#sourceTag');
          }
          updateTooltips();

          $('form#new-tag-synonym').submit(function () {
              $.ajax({
                  url: $(this).attr('action'),
                  data: $(this).serialize(),
                  error: function (request, message) {
                      StackExchange.helpers.showErrorMessage($('form#new-tag-synonym'), request.responseText);
                  },
                  success: function (data) {
                      $('.synonyms table').append(data);
                      updateTooltips();
                  },
                  type: 'POST'
              });
              return false;
          });

          $(document).on('click', '.synonyms table td.voting a', function () {
              var id = getId(this);
              var tr = $(this).parents('tr');
              var votetype = 'up';
              if ($(this).attr('class').match(/down/)) {
                  votetype = 'down';
              }

              $.ajax({
                  url: '/tag-synonym/' + votetype + 'vote/' + id,
                  data: { fkey: StackExchange.options.user.fkey },
                  error: function (request, message) {
                      StackExchange.helpers.showErrorMessage(tr, request.responseText);
                  },
                  success: function (result) { applyResult(result, tr); },
                  type: 'POST',
                  dataType: 'json'
              });

              return false;
          });

      });
  </script>
